Module OCamlMake_Common_Property.Container.LinkedList

module LinkedList: Container_LinkedList
type 'a t
Type of linked lists.
type 'a handle
A handle is a 'pointer' to an element of a linked list.

It allows removing the element in O(1) complexity.

val create : unit -> 'a t
Creates a new linked list.
val add : 'a t -> 'a -> 'a handle
Adds an element to the linked list.
val get_value : 'a handle -> 'a
Returns the value stored under a handle.
val length : 'a t -> int
Returns the size of the linked list.
val remove : 'a handle -> unit
Removes the element stored under the given handle.

Once the handle has been removed, it must not be used again.

val get_first : 'a t -> 'a
Returns the first element in the list.
val get_last : 'a t -> 'a
Returns the last element in the list.
val remove_first : 'a t -> unit
Removes the first element of the list.
val remove_last : 'a t -> unit
Removes the last element of the list.
val of_list : 'a list -> 'a t
Creates a linked list from a list.
val of_iterable : 'a Utils_Iterable.t -> 'a t
Creates a linked list from an Utils_Iterable.t.
val to_iterable : 'a t -> 'a Utils_Iterable.t
Alias for Container_LinkedList.to_forward_iterable.
val to_forward_iterable : 'a t -> 'a Utils_Iterable.t
Returns an iterable that lists all the element in the linked list.
val to_backward_iterable : 'a t -> 'a Utils_Iterable.t
Returns an iterable that lists all the element in the linked list, in reverse order.
val to_list : 'a t -> 'a list
Alias for Container_LinkedList.to_forward_list.
val to_forward_list : 'a t -> 'a list
Returns a list containing all the element in the linked list.
val to_backward_list : 'a t -> 'a list
Returns a list containing all the element in the linked list, in reverse order.
