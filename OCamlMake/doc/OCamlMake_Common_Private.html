Module OCamlMake_Common_Private

module OCamlMake_Common_Private: sig .. end
Helper functions to deal with private files (intermediary build files).
val private_folder : string
Name of the private folder that will contain intermediary build files.
val is_private : Utils_File.t -> bool
Returns true if the given file is in the private folder.
val is_public : Utils_File.t -> bool
Returns true if the given file is not in the private folder.
val to_public : Utils_File.t -> Utils_File.t
Returns the public Utils_File.t that corresponds to the given private file.

Fails if the given file is already a public file.

val to_private : Utils_File.t -> Utils_File.t
Returns the private Utils_File.t that corresponds to the given private file.

Fails if the given file is already a private file.
